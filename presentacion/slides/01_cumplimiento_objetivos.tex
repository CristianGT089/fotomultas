\section{Cumplimiento de objetivos}

% ==================== SLIDE: OBJETIVO GENERAL ====================
% CATALINA: Lee el objetivo general, enfatiza "registro y trazabilidad" (~40s)
\begin{frame}{Objetivo general}
    \begin{exampleblock}{Objetivo general}
        Desarrollar un prototipo software tecnológico que facilite el registro y la trazabilidad de los estados en el proceso de fotocomparendos en Bogotá, mediante la aplicación de tecnologías de redes distribuidas, para el fortalecimiento de la integridad y autenticidad de la información reduciendo los riesgos asociados a su confidencialidad.
    \end{exampleblock}
\end{frame}

% ==================== SLIDE: OBJETIVOS ESPECÍFICOS ====================
% CATALINA: Explica cada objetivo específico brevemente (~50s)
\begin{frame}{Objetivos específicos}
    \begin{enumerate}
        \item \textbf{Analizar} el proceso actual de registro de fotocomparendos a partir del marco jurídico, regulatorio e informes de auditoría, para identificar vulnerabilidades, requisitos funcionales y no funcionales.

        \vspace{0.8em}

        \item \textbf{Desarrollar} un prototipo con arquitectura híbrida basado en blockchain permisionado (Hyperledger Fabric) y blockchain público (Ethereum), integrando almacenamiento distribuido mediante IPFS.

        \vspace{0.8em}

        \item \textbf{Evaluar} la viabilidad técnica y funcional mediante un plan de pruebas que incluya inmutabilidad, trazabilidad, rendimiento y verificación de integridad de documentos.
    \end{enumerate}
\end{frame}
